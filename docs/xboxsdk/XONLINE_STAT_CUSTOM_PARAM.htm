XONLINE_STAT_CUSTOM_PARAM

Contains parameter information for a title-specific statistics update procedure.

typedef struct{
    XONLINE_STAT_PARAM_TYPE type;
    union {
        BYTE bValue;
        WORD wValue;
        LONG wValue;
        LONGLONG wValue;
        double dValue;
        lpString LPCWSTR;
        XUID xuidValue;
    } ;
} XONLINE_STAT_CUSTOM_PARAM, *PXONLINE_STAT_CUSTOM_PARAM;
