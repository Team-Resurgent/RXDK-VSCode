XACT_SOUNDCUE_PROPERTIES

typedef struct _XACT_SOUNDCUE_PROPERTIES{
    DWORD dwFlags;
    DWORD dwPriority;
    LONG lVolume;
    LONG lPitch;
    DWORD dwLayer;
    DWORD dwCategory;
    LONG lParametricEQGain;
    DWORD dwParametricEQQ;
    DWORD dwParametricEQFc;
    DWORD dwLoopCount;
    DWORD dwTrackCount;
    DWORD dwSoundIndex;
    DWORD dwWaveIndex;
    DWORD dwLength;
    LONG lI3DL2Volume;
    LONG lLFEVolume;
    DWORD dwInsideConeAngle;
    DWORD dwOutsideConeAngle;
    LONG lConeOutsideVolume;
    DWORD dwMode;
    FLOAT flMinDistance;
    FLOAT flMaxDistance;
    FLOAT flDistanceFactor;
    FLOAT flRolloffFactor;
    FLOAT flDopplerFactor;
} XACT_SOUNDCUE_PROPERTIES, *PXACT_SOUNDCUE_PROPERTIES;

Members

dwFlags
Flags indicating attributes of the cue. Can be zero or the following value:
Value Description
XACT_FLAG_SOUNDCUE_PROPERTIES_3D The cue is a 3D cue.
dwPriority
Priority
lVolume
Sound volume (dB * 100)
lPitch
Pitch (semitone * 4096 / 12)
dwLayer
Layer
dwCategory
Category
lParametricEQGain
Parametric EQ gain [-8192, 32767]
dwParametricEQQ
Parametric EQ Q [0, 4]
dwParametricEQFc
Parametric EQ frequency
dwLoopCount
Loop count (highest loop count of all tracks). For a sound with an infinitely looped track, this value will be XACT_LOOP_INFINITE.
dwTrackCount
Number of tracks
dwSoundIndex
Sound variation
dwWaveIndex
Wave variation
dwLength
Cue length, in milliseconds. For an infinitely looping cue, this value will be 0xFFFFFFFF. Not valid for cues that use WMA waves.
lI3DL2Volume
I3DL2 volume send (dB * 100)
lLFEVolume
LFE volume send (dB * 100)
dwInsideConeAngle
Buffer inside cone angle
dwOutsideConeAngle
Buffer outside cone angle
lConeOutsideVolume
Volume outside the cone
dwMode
3D processing mode
flMinDistance
Minimum distance value
flMaxDistance
Maximum distance value
flDistanceFactor
Distance factor
flRolloffFactor
Rolloff factor
flDopplerFactor
Doppler factor
